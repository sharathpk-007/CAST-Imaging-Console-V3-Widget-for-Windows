def open_folder(path_to_open):
    try:
        normalized_path = os.path.normpath(path_to_open)
        if os.path.exists(normalized_path):
            if os.name == 'nt': os.startfile(normalized_path)
            elif os.name == 'posix':
                if subprocess.call(['which', 'xdg-open'], stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL) == 0: subprocess.call(['xdg-open', normalized_path])
                elif subprocess.call(['which', 'open'], stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL) == 0: subprocess.call(['open', normalized_path])
                else: messagebox.showwarning("Cannot Open", f"Could not find xdg-open or open.")
            else: messagebox.showwarning("Cannot Open", f"Unsupported OS.")
        else: messagebox.showerror("Error", f"Path does not exist:\n{normalized_path}")
    except Exception as e: messagebox.showerror("Error", f"Could not open folder: {path_to_open}\n{str(e)}")

def parse_index_xml(xml_file_path):
    entries = []
    try:
        tree = ET.parse(xml_file_path)
        root_xml = tree.getroot()
        for entry_elem in root_xml.findall('.//entry'):
            guid = entry_elem.get('key')
            app_name = entry_elem.text
            if guid and app_name:
                entries.append({'guid': guid.strip(), 'name': app_name.strip()})
    except ET.ParseError as e:
        messagebox.showerror("XML Parse Error", f"Error parsing {os.path.basename(xml_file_path)}:\n{e}")
        raise
    return entries

def db_clear_and_recreate_results_tree():
    global db_results_tree, db_results_tree_scrollbar_y, db_results_tree_scrollbar_x, results_frame

    if not results_frame: # Ensure parent frame exists
        print("Error: results_frame not initialized for db_results_tree. Cannot recreate.")
        return None

    # Destroy existing tree and scrollbars if they exist
    if db_results_tree is not None and db_results_tree.winfo_exists():
        db_results_tree.destroy()
    if db_results_tree_scrollbar_y is not None and db_results_tree_scrollbar_y.winfo_exists():
        db_results_tree_scrollbar_y.destroy()
    if db_results_tree_scrollbar_x is not None and db_results_tree_scrollbar_x.winfo_exists():
        db_results_tree_scrollbar_x.destroy()

    # Recreate scrollbars and tree, assign to globals
    db_results_tree_scrollbar_y = ttk.Scrollbar(results_frame, orient="vertical")
    db_results_tree_scrollbar_x = ttk.Scrollbar(results_frame, orient="horizontal")
    db_results_tree = ttk.Treeview(results_frame, 
                                   yscrollcommand=db_results_tree_scrollbar_y.set, 
                                   xscrollcommand=db_results_tree_scrollbar_x.set, 
                                   show="headings") 
    db_results_tree_scrollbar_y.config(command=db_results_tree.yview)
    db_results_tree_scrollbar_x.config(command=db_results_tree.xview)
    
    db_results_tree_scrollbar_y.pack(side=tk.RIGHT, fill=tk.Y)
    db_results_tree_scrollbar_x.pack(side=tk.BOTTOM, fill=tk.X)
    db_results_tree.pack(fill="both", expand=True)
    return db_results_tree # Return the new tree instance

def connect_db():
    global db_connection, PSYCOPG2_AVAILABLE, db_entries, db_status_label_var, root
    global db_connect_button, db_disconnect_button, db_execute_query_button, db_schema_combobox
    global db_table_list
    
    if not PSYCOPG2_AVAILABLE: messagebox.showerror("Missing Library", "psycopg2 not installed."); return
    if db_connection: disconnect_db() 
    
    if PSYCOPG2_AVAILABLE: db_clear_and_recreate_results_tree()
    db_table_list.delete(0, tk.END)
    try:
        host, port, dbname, user, password = (db_entries[k].get() for k in ['host', 'port', 'database_name', 'username', 'password'])
        if not all([host, port, dbname, user]): messagebox.showerror("Input Error", "Host, Port, Database Name, and Username are required."); return
        
        db_status_label_var.set(f"Status: Connecting to {dbname} on {host}:{port}..."); root.update_idletasks()
        db_connection = psycopg2.connect(host=host, port=port, dbname=dbname, user=user, password=password, connect_timeout=5)
        db_status_label_var.set(f"Status: Connected to {dbname} on {host}:{port}")
        db_connect_button.config(state=tk.DISABLED); db_disconnect_button.config(state=tk.NORMAL); db_execute_query_button.config(state=tk.NORMAL)
        db_schema_combobox.config(state="readonly") 
        if db_schema_combobox.get(): load_tables_for_schema()

    except psycopg2.Error as e:
        db_status_label_var.set("Status: Connection Failed"); messagebox.showerror("Database Error", f"Could not connect: {e}")
        if db_connection: db_connection.close(); db_connection = None
    except Exception as e:
        db_status_label_var.set("Status: Error"); messagebox.showerror("Error", f"An unexpected error: {e}")
        if db_connection: db_connection.close(); db_connection = None

def disconnect_db():
    global db_connection, db_status_label_var, db_connect_button, db_disconnect_button
    global db_execute_query_button, db_table_list, db_schema_combobox
    if db_connection:
        try: db_connection.close()
        except psycopg2.Error: pass 
        db_connection = None
    db_status_label_var.set("Status: Disconnected")
    db_connect_button.config(state=tk.NORMAL); db_disconnect_button.config(state=tk.DISABLED); db_execute_query_button.config(state=tk.DISABLED)
    db_table_list.delete(0, tk.END)
    if PSYCOPG2_AVAILABLE: 
         db_clear_and_recreate_results_tree() 
    db_schema_combobox.set(''); db_schema_combobox.config(state="disabled")

def load_tables_for_schema(event=None):
    global db_connection, PSYCOPG2_AVAILABLE, db_selected_schema, db_table_list
    if not db_connection or not PSYCOPG2_AVAILABLE: return
    schema_name = db_selected_schema.get()
    if not schema_name: return
    db_table_list.delete(0, tk.END)
    if PSYCOPG2_AVAILABLE:
        db_clear_and_recreate_results_tree() 
    try:
        with db_connection.cursor() as cur:
            query = sql.SQL("SELECT table_name FROM information_schema.tables WHERE table_schema = %s ORDER BY table_name;")
            cur.execute(query, (schema_name,))
            tables = [row[0] for row in cur.fetchall()]
            if tables:
                for table in tables: db_table_list.insert(tk.END, table)
            else:
                db_table_list.insert(tk.END, "(No tables found)")
    except psycopg2.Error as e: messagebox.showerror("DB Error", f"Error fetching tables for schema '{schema_name}':\n{e}")
    except Exception as e: messagebox.showerror("Error", f"Unexpected error fetching tables:\n{e}")

def display_table_schema(event=None): 
    global db_connection, PSYCOPG2_AVAILABLE, db_table_list, db_selected_schema, db_results_tree
    if not db_connection or not PSYCOPG2_AVAILABLE: return
    selected_indices = db_table_list.curselection()
    if not selected_indices: return
    
    table_name = db_table_list.get(selected_indices[0])
    schema_name = db_selected_schema.get()
    if not table_name or not schema_name or table_name == "(No tables found)": return

    current_tree = db_clear_and_recreate_results_tree() 
    if not current_tree: return
    current_tree['show'] = 'headings'

    try:
        with db_connection.cursor() as cur:
            query = sql.SQL("SELECT column_name, data_type, is_nullable FROM information_schema.columns WHERE table_schema = %s AND table_name = %s ORDER BY ordinal_position;")
            cur.execute(query, (schema_name, table_name))
            columns_info = cur.fetchall()
            if columns_info:
                col_names = ["Column Name", "Data Type", "Nullable"]
                current_tree["columns"] = col_names; current_tree["displaycolumns"] = col_names
                for col in col_names: current_tree.heading(col, text=col); current_tree.column(col, width=150, anchor="w")
                for row_data in columns_info: current_tree.insert("", tk.END, values=row_data)
            else:
                current_tree["columns"] = ("Status",); current_tree.heading("Status", text="Status")
                current_tree.insert("",tk.END, values=(f"No column info for {schema_name}.{table_name}",))
    except psycopg2.Error as e: messagebox.showerror("DB Error", f"Error fetching schema for {schema_name}.{table_name}:\n{e}")
    except Exception as e: messagebox.showerror("Error", f"Unexpected error fetching table schema:\n{e}")

def execute_db_query():
    global db_connection, PSYCOPG2_AVAILABLE, db_query_text, db_results_tree
    if not db_connection or not PSYCOPG2_AVAILABLE: messagebox.showerror("Error", "Not connected to database."); return
    query_str = db_query_text.get("1.0", tk.END).strip()
    if not query_str: messagebox.showwarning("Input Error", "Query cannot be empty."); return
    if not query_str.lower().lstrip().startswith("select"): messagebox.showerror("Query Error", "Only SELECT statements are allowed."); return
        
    current_tree = db_clear_and_recreate_results_tree()
    if not current_tree: return
    current_tree['show'] = 'headings'

    try:
        with db_connection.cursor() as cur:
            cur.execute(query_str) 
            if cur.description: 
                col_names = [desc[0] for desc in cur.description]
                current_tree["columns"] = col_names; current_tree["displaycolumns"] = col_names 
                for col in col_names: current_tree.heading(col, text=col); current_tree.column(col, width=100, anchor="w")
                rows_fetched = 0; max_rows_to_display = 500 
                for row_data in cur:
                    current_tree.insert("", tk.END, values=row_data); rows_fetched += 1
                    if rows_fetched >= max_rows_to_display:
                        current_tree.insert("", tk.END, values=("...",)*(len(col_names))); current_tree.insert("", tk.END, values=(f"(Displayed first {max_rows_to_display} rows)",)*(len(col_names))); break
                if rows_fetched == 0: current_tree.insert("",tk.END, values=("(No rows returned)",)*(len(col_names) if col_names else 1) )
            else: 
                status_message = f"Query executed. Rows affected: {cur.rowcount}" if cur.rowcount != -1 else "Query executed."
                current_tree["columns"] = ("Status",); current_tree.heading("Status", text="Status"); current_tree.insert("",tk.END, values=(status_message,))
        db_connection.commit() 
    except psycopg2.Error as e: messagebox.showerror("Query Execution Error", f"Error executing query:\n{e}"); db_connection.rollback() 
    except Exception as e: messagebox.showerror("Error", f"Unexpected error executing query:\n{e}"); 
    if db_connection: db_connection.rollback()

def parse_config_all_conf(conf_path):
    raw_config, resolved_config = {}, {}; 
    expected_keys = ["imaging.root", "imaging.viewer.log.dir", "INSTALL_DIR", "DATA_DIR", "AIP_INSTALLDIR", "SHARED_FOLDER", "DELIVERY_FOLDER", "SOURCES_FOLDER_LOCATION"]
    extracted_values = {key: "" for key in expected_keys}
    try:
        with open(conf_path, "r", encoding="utf-8") as f:
            for line in f:
                line = line.strip(); 
                if not line or line.startswith("#"): continue
                if "=" in line:
                    key, value = line.split("=", 1); key = key.strip(); value = value.strip().strip('"').strip("'")
                    pk = key[4:] if key.startswith("env.") or key.startswith("sys.") else key; raw_config[pk] = value; 
                    if key not in raw_config: raw_config[key] = value
        var_pattern = re.compile(r"\$\{([^}]+)\}|\%([^%]+)\%")
        def resolve(val, cfg):
            for _ in range(10):
                orig_val, matches = val, list(var_pattern.finditer(val)); 
                if not matches: break
                for m in reversed(matches): vk = (m.group(1) or m.group(2)).split(':',1)[-1]; 
                if vk in cfg: val = val[:m.start()] + cfg[vk] + val[m.end():]
                if val == orig_val and matches: break
            return val
        for k, v in raw_config.items(): resolved_config[k] = resolve(v, raw_config)
        for k, v in resolved_config.items(): resolved_config[k] = resolve(v, resolved_config)
        for ek in expected_keys: extracted_values[ek] = resolved_config.get(ek, resolve(raw_config.get(ek, ""), resolved_config) if raw_config.get(ek) else "")
    except FileNotFoundError: messagebox.showerror("Error", f"Config file not found: {conf_path}"); return {k: "" for k in extracted_values}
    except Exception as e: messagebox.showerror("Error", f"Failed to parse config: {str(e)}"); return {k: "" for k in extracted_values}
    return extracted_values

    def update():
        for service_name in SERVICE_NAMES:
            display_status = get_service_status(service_name); tag = "NotFound"; 
            if display_status == "Running": tag = "Running"
            elif display_status == "Stopped": tag = "Stopped"
            elif display_status == "Stopping": tag = "StopPending"
            if tree.exists(service_name): tree.item(service_name, values=(service_name, display_status), tags=(tag,))
        root.after(7000, update)
    update()

def show_log(service_name):

    try:
        with open(log_path, "r", encoding="utf-8", errors="ignore") as file:
            lines = file.readlines(); errors = [l for l in lines if 'error' in l.lower()]
            log_display.insert(tk.END, f"--- {service_name} ({os.path.basename(log_path)}) ---\n")
            if errors: log_display.insert(tk.END, "Errors:\n\n" + ''.join(errors))
            elif lines: log_display.insert(tk.END, "Last 100 lines:\n\n" + ''.join(lines[-100:]))
            else: log_display.insert(tk.END, "Log empty.")
    except Exception as e: log_display.insert(tk.END, f"Failed to read log: {log_path}\nError: {str(e)}")
